第一章

最简单的程序

#include <stdio.h>  /*文件包含*/

int main ( )  /*主函数 */

{                                  

printf ("This is a C program.\n"); 

return 0; //当main函数执行结束前将整数0作为函数值

}

说明： main-主函数名， void-函数类型

• 每个C程序必须有一个主函数main

• { }是函数开始和结束的标志，不可省

• 每个C语句以分号结束

• 使用标准库函数时应在程序开头一行写：

#include <stdio.h>

华氏温度到摄氏温度


#include <stdio.h>

int main ( )

{

float f, c;//定义f和c为单精度浮点型变量

f=32;//确定f的值

c=(5.0/9)\*(f-32);//利用公式计算c的值

printf("f=%f\nc=%f\n", f, c);//按行输出f和c的值

return 0;

}

数据类型与计算机系统结构密不可分

✓ 16、32位编译器：int、short型变量分配2个字节空间

✓ 64位编译器：int型变量分配4个字节空间，short2个字节

void main( ) {
short int a=32767, b;
b=a+1;
printf("a=%d b=%d\n", a, b);
}

转义字符

用%m.nf指定数据宽度和小数位数

printf(“%15.13f\n”, a/7);

例：将一任意十进制数转换成三进制并输出

#include<stdio.h>
int main( ) {
  int i=0, x, d[20];
  scanf("%d", &x);
  while(x>0) {
    d[i++]=x%3;
    x=x/3;
  }
for(i--; i>=0; i--) printf("%d", d[i]);
printf("\n"); 
return 0;
}
结果：
49
1211

C语言对外部数据的组织和处理

#include <stdio.h>  
#include<stdlib.h>  
  
int main( )   
{   
    FILE *fp;   
    char ch;  
  
    // 写入文件  
    if((fp=fopen("f:/CAI/C/tmp/myfile.dat","w"))==NULL)   
    {   
        printf("打开文件失败\n");   
        exit(0);   
    }  
    while((ch=getchar())!='@') fputc(ch, fp);   
    fclose(fp);  
  
    // 从文件中读取并打印内容  
    if((fp=fopen("f:/CAI/C/tmp/myfile.dat","r"))==NULL)   
    {   
        printf("打开文件失败\n");   
        exit(0);   
    }  
    while((ch=fgetc(fp))!=EOF) putchar(ch);   
    printf("\n");  
    fclose(fp);  
  
    return 0;  
}

选择结构

代码：

#include "stdio.h"  
  
int main() {  
    int week, num;  
    int flag = 0; // flag初始化为0  
    printf("请输入今天是周几(1～7)、您的车牌尾号(0～9):");  
    scanf("%d%d", &week, &num); // 输入时间和车牌尾号  
    switch (num) {  
        case 0:  
        case 1:  
            if (week == 1) {  
                flag = 1;  
            }  
            break;  
        case 2:  
        case 6:  
            if (week == 2) {  
                flag = 1;  
            }  
            break;  
        case 3:  
        case 7:  
            if (week == 3) {  
                flag = 1;  
            }  
            break;  
        case 4:  
        case 8:  
            if (week == 4) {  
                flag = 1;  
            }  
            break;  
        case 5:  
        case 9:  
            if (week == 5) {  
                flag = 1;  
            }  
            break;  
        default: // 如果车牌尾号不在0-9之间，输出错误信息并退出程序  
            printf("您输入的车牌尾号不正确！\n");  
            return -1; // 或者使用return语句返回一个错误码，以表示程序异常退出。例如：return -1;  
    }  
    if (flag == 1) { // 根据flag的值输出车辆限行信息或不限行信息  
        printf("对不起，您的车辆今天限行！\n");  
    } else {  
        printf("恭喜您，您的车辆今天不限行！\n");  
    }  
    return 0; // 表示程序正常退出。也可以使用return语句返回一个整数值表示程序的退出状态。例如：return 0; // 表示程序正常退出。return -1; // 表示程序异常退出。
