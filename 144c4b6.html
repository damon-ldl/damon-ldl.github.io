一.通过anaconda实现本地访问chatglm

1.安装python库

pip install gradio_client

2.在anaconda的jupyter实现代码

from gradio_client import Client

import json

def call_internlm_cgat6b(input_):
    """
    input_: 当前输入
    return: 当前输出
    """
    client = Client("https://modelscope.cn/api/vl/studio/AI-ModelScope/Internlm-chat-7b/gradio/")
    result = client.predict(input_,fn_index=0)
    with open(result[1],'r',encoding='utf-8') as f:
        data = json.load(f)
    return data[-1][-1]

3.初步体验

call_internlm_chat7b('你好')

完整代码如下：

import json

# 从resume.json文件中读取简历信息
with open('resume.json', 'r', encoding='utf-8') as resume_file:
    resume = json.load(resume_file)

# 从job_requirements.json文件中读取职位要求信息
with open('job_requirements.json', 'r', encoding='utf-8') as job_requirements_file:
    job_requirements = json.load(job_requirements_file)

output = '''
{
"推荐岗位": "RD还是PM还是UI(三选一)",
"简历得分": "分数范围0-100",
"得分依据": "根据简历和岗位要求给出详细的得分依据，量化每一部分得分"
}
'''

# 将简历和职位要求添加到文本中
text = f'请评估以下简历和岗位匹配度：\n{json.dumps(resume, indent=4, ensure_ascii=False)}\n\n岗位要求：\n{json.dumps(job_requirements, indent=4, ensure_ascii=False)}\n输出的结果形式如下：\n{output}，输出格式为json格式'

# 调用call_internlm_chat7b函数
response = call_internlm_chat7b(text)

# 打印结果
print(response)
